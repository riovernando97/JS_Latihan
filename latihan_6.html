      // Fungsi konversi suhu
      // Celcius To Fahrenheit
      function celciusFahrenheit(celcius) {
        return (celcius * 9) / 5 + 32;
      }

      // Fahrenheit to Celcius
      function fahrenheitCelcius(fahrenheit) {
        return ((fahrenheit - 32) * 5) / 9;
      }

      //  Celcius to Kelvin
      function celciusKelvin(celcius) {
        return celcius + 273.15;
      }

      // Kelvin to Celcius
      function kelvinCelcius(kelvin) {
        return kelvin - 273.15;
      }

      // Fahrenheit to Kelvin
      function fahrenheitKelvin(fahrenheit) {
        return ((fahrenheit - 32) * 5) / 9 + 273.15;
      }

      // Kelvin to Fahrenheit
      function kelvnFahrenheit(kelvin) {
        return ((kelvin - 273.15) * 9) / 5 + 32;
      }

      let celcius = 34;
      let fahrenheit = 22;
      let kelvin = 40.14;

      // konv celcius ke fahrenheit
      document.write(`${celcius} °C = ${celciusFahrenheit(celcius)} °F`);

      // konv fahrenheit ke celcius
      document.write(`${fahrenheit} °F = ${fahrenheitCelcius(fahrenheit)} °F`);

      // konv celcius ke kelvin
      document.write(`${celcius} °C = ${celciusKelvin(celcius)} °K`);

      // konv kelvin ke celcius
      document.write(`${kelvin} °K = ${kelvinCelcius(kelvin)} °C`);

      // konv fahrenheit ke kelvin
      document.write(`${fahrenheit} °F = ${fahrenheitKelvin(fahrenheit)} °K`);

      // konv kelvin ke fahrenheit
      document.write(`${kelvin} °K = ${kelvnFahrenheit(kelvin)} °F`);
